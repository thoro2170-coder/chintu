def linear_search(accounts, target):
    for i, acc in enumerate(accounts):
        if acc == target:
            return i
    return -1


def binary_search(accounts, target):
    sorted_list = sorted(accounts)
    low = 0
    high = len(sorted_list) - 1

    while low <= high:
        mid = (low + high) // 2
        if sorted_list[mid] == target:
            return mid
        elif sorted_list[mid] < target:
            low = mid + 1
        else:
            high = mid - 1
    return -1


# ------- Main Program -------
accounts = [1025, 1001, 1050, 1100, 1085]

try:
    search_id = int(input("Enter Customer Account ID to search: "))
except ValueError:
    print("Invalid Input, Please enter a number")
    exit()

# Linear Search
lin_pos = linear_search(accounts, search_id)
if lin_pos != -1:
    print(f"[Linear Search] Account ID {search_id} found at position {lin_pos}")
else:
    print("Account ID not found (Linear Search)")

# Binary Search
bin_pos = binary_search(accounts, search_id)
if bin_pos != -1:
    print(f"[Binary Search] Account ID {search_id} found at position {bin_pos}")
else:
    print("Account ID not found (Binary Search)")
